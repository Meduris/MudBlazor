using Microsoft.Extensions.Localization;
using System.Threading;
using MudBlazor;

internal class CustomInterceptor : ILocalizationInterceptor
{
    public LocalizedString Handle(string key, IStringLocalizer defaultEnglishLocalizer, MudLocalizer customMudLocalizer)
    {
        var currentCulture = Thread.CurrentThread.CurrentUICulture.Parent.TwoLetterISOLanguageName;
        if (customMudLocalizer == null || currentCulture.Equals("en", StringComparison.InvariantCultureIgnoreCase)
        {
            switch (key)
            {
                case "MudDataGrid.Clear":
                    return new LocalizedString(key, "Reset");
                case "MudDataGrid.Apply":
                    return new LocalizedString(key, "Confirm");
            }
            
            return defaultEnglishLocalizer[key];
        }
        
        var res = customMudLocalizer[key];
        return res.ResourceNotFound ? defaultEnglishLocalizer[key] : res;
    }
}
